49 def getEvalStats(self, predictedVarName=None) -> TEvalStats: 

50 if predictedVarName is None: 

51 if len(self.evalStatsByVarName) != 1: 

52 raise Exception(f"Must provide name of predicted variable name, as multiple variables were predicted {list(self.evalStatsByVarName.keys())}") 

53 else: 

54 predictedVarName = next(iter(self.evalStatsByVarName.keys())) 

55 evalStats = self.evalStatsByVarName.get(predictedVarName) 

56 if evalStats is None: 

57 raise ValueError(f"No evaluation data present for '{predictedVarName}'; known output variables: {list(self.evalStatsByVarName.keys())}") 

58 return evalStats

64 :return: a DataFrame containing evaluation metrics

66 statsDicts = []

89class PredictorModelEvaluator(MetricsDictProvider, Generic[TEvalData], ABC): 

90 def __init__(self, data: InputOutputData, testData: InputOutputData = None, dataSplitter: DataSplitter = None, 

91 testFraction: float = None, randomSeed=42, shuffle=True): 

92 """ 

93 Constructs an evaluator with test and training data. 

94 Exactly one of the parameters {testData, dataSplitter, testFraction} must be given 

95 

96 :param data: the full data set, or, if testData is given, the training data 

97 :param testData: the data to use for testing/evaluation; if None, must specify either dataSplitter testFraction or dataSplitter 

98 :param dataSplitter: [if testData is None] a splitter to use in order to obtain; if None, must specify either testData or testFraction 

99 :param testFraction: [if testData is None, dataSplitter is None] the fraction of the data to use for testing/evaluation; 

100 if None, must specify either testData or dataSplitter 

101 :param randomSeed: [if data is None, dataSplitter is None] the random seed to use for the fractional split of the data 

102 :param shuffle: [if data is None, dataSplitter is None] whether to randomly (based on randomSeed) shuffle the dataset before 

103 splitting it 

104 """ 

105 if (testData, dataSplitter, testFraction).count(None) != 2: 

106 raise ValueError("Exactly one of {testData, dataSplitter, testFraction} must be given") 

107 if testFraction is not None: 

108 if not 0 <= testFraction <= 1: 

109 raise Exception(f"testFraction has to be None or within the interval [0, 1]. Instead got: {testFraction}") 

110 if testData is None: 

111 if dataSplitter is None: 

112 dataSplitter = DataSplitterFractional(1 - testFraction, shuffle=shuffle, randomSeed=randomSeed) 

113 self.trainingData, self.testData = dataSplitter.split(data) 

114 else: 

115 self.trainingData = data 

116 self.testData = testData

118 def evalModel(self, model: PredictorModel, onTrainingData=False) -> TEvalData: 

119 """ 

120 Evaluates the given model 

121 

122 :param model: the model to evaluate 

123 :param onTrainingData: if True, evaluate on this evaluator's training data rather than the held-out test data 

124 :return: the evaluation result 

125 """ 

126 data = self.trainingData if onTrainingData else self.testData 

127 return self._evalModel(model, data)
